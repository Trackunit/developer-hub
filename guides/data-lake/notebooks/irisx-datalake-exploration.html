# Analyze running hours over time running_hours_query = """ SELECT asset_id, timestamp, value as running_hours FROM irisx.insights.runninghours_hourly WHERE asset_id = '12345' AND timestamp >= current_date() - interval 7 days ORDER BY timestamp """ hours_df = pd.read_sql(running_hours_query, connection) # Create visualization fig = px.line(hours_df, x='timestamp', y='running_hours', title='Running Hours Over Time') fig.show()

üìä Fleet Performance Dashboard

# Fleet-wide performance metrics fleet_query = """ SELECT a.manufacturer, COUNT(*) as asset_count, AVG(ih.value) as avg_running_hours, AVG(if.value) as avg_fuel_consumption FROM irisx.asset.asset a LEFT JOIN irisx.insights.runninghours_latest ih ON a.asset_id = ih.asset_id LEFT JOIN irisx.insights.fuelconsumption_latest if ON a.asset_id = if.asset_id GROUP BY a.manufacturer ORDER BY asset_count DESC """ fleet_df = pd.read_sql(fleet_query, connection) # Create bar chart fig = px.bar(fleet_df, x='manufacturer', y='asset_count', title='Assets by Manufacturer') fig.show()
